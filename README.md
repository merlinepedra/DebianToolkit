**Install:**
```
root@debian:~# apt-get install -y git

root@debian:~# cd ~

root@debian:~# git clone https://github.com/d4t4s3c/DebianToolkit.git

root@debian:~# cd DebianToolkit

root@debian:~# chmod +x DebianToolkit.sh

root@debian:~# ./DebianToolkit. sh
```
